response.setHeader("refresh","2") ;      //页面刷新（2秒） 

response.setHeader("refresh","3;URL=hello.htm") ;        // 3秒后跳转到hello.htm页面，这种跳转时而有错

<META HTTP-EQUIV="refresh" CONTENT="3;URL=hello.htm"> //使用HTML方式进行跳转

<jsp:forward page="hello.htm"/>//服务器端跳转，跳转后地址栏不变，属于无条件的跳转，执行之后立刻跳转，跳转之后的语句不会执行。可以传递request属性

response.sendRedirect("hello.htm") ;//客户端跳转，跳转后地址栏改变，不可以传递request属性，是在所有语句执行完之后再跳转。
